# LeetCode- 70. Climbing Stairs
# https://leetcode.com/problems/climbing-stairs/

class Solution(object):
    def climbStairs(self, n):
        """
        :type n: int
        :rtype: int
        """
        #THis will be solved by dynamic programming

        one,two=1,1
        for i in range(n-1):
            temp=one
            one=one+two
            two=temp

        return one
